```java
package com.bulldog.domain.value_objects;

import java.util.regex.Pattern;
import com.bulldog.domain.exceptions.InvalidEmailException;

/**
 * Value Object for Email
 * Immutable and enforces email format rules
 */
public record Email(String value) {

    private static final Pattern EMAIL_PATTERN = Pattern.compile(
        "^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+\\.[A-Z|a-z]{2,}$"
    );

    /**
     * Constructor with validation
     */
    public Email {
        if (value == null || value.isBlank()) {
            throw new InvalidEmailException("Email cannot be null or empty");
        }
        if (!EMAIL_PATTERN.matcher(value).matches()) {
            throw new InvalidEmailException("Invalid email format: " + value);
        }
    }

    /**
     * Create Email from string
     */
    public static Email of(String email) {
        return new Email(email);
    }

    @Override
    public String toString() {
        return value;
    }
}
```

```java
    }
```

```java
// filepath: bulldog-api/src/main/java/com/bulldog/domain/entities/User.java

package com.bulldog.domain.entities;

import com.bulldog.domain.value_objects.Email;
import com.bulldog.domain.value_objects.Password;
import com.bulldog.domain.exceptions.InvalidUserException;
import lombok.*;
import java.time.LocalDateTime;
import java.util.*;

/**
 * User Domain Entity
 * Core business logic for user management
 * Independence from framework/persistence
 */
@NoArgsConstructor(access = AccessLevel.PROTECTED)
@AllArgsConstructor(access = AccessLevel.PRIVATE)
@Getter
public class User {

    // Aggregates root ID
    private Long id;

    // Value Objects
    private Email email;
    private Password password;

    // User properties
    private String name;
    private Integer age;
    private Boolean isActive;

    // Audit
    private LocalDateTime createdAt;
    private LocalDateTime updatedAt;

    // Bank accounts collection (child entities)
    private Set<BankAccount> bankAccounts;

    /**
     * Create new user (factory method)
     */
    public static User create(String name, Email email, Password password, Integer age) {
        validateUserCreation(name, email, password);

        User user = new User();
        user.id = null; // Generated by persistence layer
        user.name = name;
        user.email = email;
        user.password = password;
        user.age = age;
        user.isActive = true;
        user.createdAt = LocalDateTime.now();
        user.updatedAt = LocalDateTime.now();
        user.bankAccounts = new HashSet<>();

        return user;
    }

    /**
     * Reconstruct user from persistence (used by repository)
     */
    public static User reconstruct(
        Long id,
        String name,
        Email email,
        Password password,
        Integer age,
        Boolean isActive,
        LocalDateTime createdAt,
        LocalDateTime updatedAt,
        Set<BankAccount> bankAccounts
    ) {
        User user = new User();
        user.id = id;
        user.name = name;
        user.email = email;
        user.password = password;
        user.age = age;
        user.isActive = isActive;
        user.createdAt = createdAt;
        user.updatedAt = updatedAt;
        user.bankAccounts = bankAccounts;

        return user;
    }

    /**
     * Update user profile (business logic)
     */
    public void updateProfile(String name, Integer age) {
        if (name == null || name.isBlank()) {
            throw new InvalidUserException("Name cannot be empty");
        }
        if (age != null && age < 0) {
            throw new InvalidUserException("Age cannot be negative");
        }

        this.name = name;
        this.age = age;
        this.updatedAt = LocalDateTime.now();
    }

    /**
     * Change password (business logic)
     */
    public void changePassword(Password newPassword) {
        if (newPassword == null) {
            throw new InvalidUserException("New password cannot be null");
        }
        this.password = newPassword;
        this.updatedAt = LocalDateTime.now();
    }

    /**
     * Deactivate user account
     */
    public void deactivate() {
        this.isActive = false;
        this.updatedAt = LocalDateTime.now();
    }

    /**
     * Activate user account
     */
    public void activate() {
        this.isActive = true;
        this.updatedAt = LocalDateTime.now();
    }

    /**
     * Add bank account
     */
    public void addBankAccount(BankAccount bankAccount) {
        if (bankAccount == null) {
            throw new InvalidUserException("Bank account cannot be null");
        }
        this.bankAccounts.add(bankAccount);
    }

    /**
     * Remove bank account
     */
    public void removeBankAccount(BankAccount bankAccount) {
        this.bankAccounts.remove(bankAccount);
    }

    /**
     * Get active bank accounts
     */
    public Set<BankAccount> getActiveBankAccounts() {
        return bankAccounts.stream()
            .filter(BankAccount::isActive)
            .collect(java.util.stream.Collectors.toSet());
    }

    /**
     * Get bank account by ID
     */
    public BankAccount getBankAccount(Long accountId) {
        return bankAccounts.stream()
            .filter(account -> account.getId().equals(accountId))
            .findFirst()
            .orElseThrow(() -> new InvalidUserException(
                "Bank account not found: " + accountId
            ));
    }

    /**
     * Validate user creation
     */
    private static void validateUserCreation(String name, Email email, Password password) {
        if (name == null || name.isBlank()) {
            throw new InvalidUserException("User name is required");
        }
        if (email == null) {
            throw new InvalidUserException("Email is required");
        }
        if (password == null) {
            throw new InvalidUserException("Password is required");
        }
    }

    /**
     * Check if user is active
     */
    public boolean isUserActive() {
        return Boolean.TRUE.equals(this.isActive);
    }

    /**
     * Get user summary
     */
    public String getSummary() {
        return String.format(
            "User(id=%d, name=%s, email=%s, active=%b)",
            id, name, email, isActive
        );
    }
}
```

```java
package com.bulldog.domain.services;

/**
 * User Domain Service
 * Encapsulates complex business logic involving User
 */
@Service
@RequiredArgsConstructor
public class UserDomainService {

    private final UserRepository userRepository;

    /**
     * Verify email is not already registered
     */
    public void verifyEmailNotRegistered(Email email) {
        if (userRepository.existsByEmail(email)) {
            throw new UserAlreadyExistsException(
                "Email already registered: " + email
            );
        }
    }

    /**
     * Get user with all details
     */
    public User getUserWithDetails(Long userId) {
        return userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException(
                "User not found: " + userId
            ));
    }

    /**
     * Verify user is active before operations
     */
    public void verifyUserActive(User user) {
        if (!user.isUserActive()) {
            throw new UserInactiveException(
                "User account is inactive: " + user.getId()
            );
        }
    }
}
```

```java
// filepath: bulldog-api/src/main/java/com/bulldog/application/services/UserApplicationService.java

package com.bulldog.application.services;

import com.bulldog.domain.entities.User;
import com.bulldog.domain.value_objects.Email;
import com.bulldog.domain.value_objects.Password;
import com.bulldog.domain.repositories.UserRepository;
import com.bulldog.domain.services.UserDomainService;
import com.bulldog.domain.exceptions.UserNotFoundException;
import com.bulldog.application.dto.requests.CreateUserRequest;
import com.bulldog.application.dto.requests.UpdateUserRequest;
import com.bulldog.application.dto.requests.ChangePasswordRequest;
import com.bulldog.application.dto.responses.UserDTO;
import com.bulldog.application.dto.responses.UserDetailDTO;
import com.bulldog.application.mappers.UserMapper;
import com.bulldog.application.ports.UserPasswordEncoder;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.List;
import java.util.stream.Collectors;

/**
 * User Application Service
 * Orchestration of use cases
 */
@Service
@RequiredArgsConstructor
@Transactional
public class UserApplicationService {

    private final UserRepository userRepository;
    private final UserDomainService userDomainService;
    private final UserMapper userMapper;
    private final UserPasswordEncoder passwordEncoder;

    /**
     * Create new user (Use Case)
     */
    public UserDTO createUser(CreateUserRequest request) {
        // Verify email not already registered
        Email email = userMapper.toEmail(request.getEmail());
        userDomainService.verifyEmailNotRegistered(email);

        // Hash password
        Password hashedPassword = passwordEncoder.encode(request.getPassword());

        // Create domain entity
        User user = User.create(
            request.getName(),
            email,
            hashedPassword,
            request.getAge()
        );

        // Save to repository
        User savedUser = userRepository.save(user);

        // Return DTO
        return userMapper.toDTO(savedUser);
    }

    /**
     * Get user by ID (Use Case)
     */
    @Transactional(readOnly = true)
    public UserDTO getUserById(Long userId) {
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("User not found: " + userId));

        userDomainService.verifyUserActive(user);

        return userMapper.toDTO(user);
    }

    /**
     * Get user details with accounts (Use Case)
     */
    @Transactional(readOnly = true)
    public UserDetailDTO getUserDetails(Long userId) {
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("User not found: " + userId));

        userDomainService.verifyUserActive(user);

        return userMapper.toDetailDTO(user);
    }

    /**
     * Get all active users (Use Case)
     */
    @Transactional(readOnly = true)
    public List<UserDTO> getAllActiveUsers() {
        return userRepository.findAllActive().stream()
            .map(userMapper::toDTO)
            .collect(Collectors.toList());
    }

    /**
     * Update user profile (Use Case)
     */
    public UserDTO updateUserProfile(Long userId, UpdateUserRequest request) {
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("User not found: " + userId));

        userDomainService.verifyUserActive(user);

        // Update domain entity
        user.updateProfile(request.getName(), request.getAge());

        // Persist changes
        userRepository.update(user);

        return userMapper.toDTO(user);
    }

    /**
     * Change user password (Use Case)
     */
    public void changePassword(Long userId, ChangePasswordRequest request) {
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("User not found: " + userId));

        userDomainService.verifyUserActive(user);

        // Verify current password
        if (!passwordEncoder.matches(request.getCurrentPassword(), user.getPassword().getHashedValue())) {
            throw new IllegalArgumentException("Current password is incorrect");
        }

        // Verify passwords match
        if (!request.getNewPassword().equals(request.getConfirmPassword())) {
            throw new IllegalArgumentException("Passwords do not match");
        }

        // Hash new password
        Password hashedPassword = passwordEncoder.encode(request.getNewPassword());

        // Update domain entity
        user.changePassword(hashedPassword);

        // Persist changes
        userRepository.update(user);
    }

    /**
     * Deactivate user account (Use Case)
     */
    public void deactivateUser(Long userId) {
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("User not found: " + userId));

        user.deactivate();
        userRepository.update(user);
    }

    /**
     * Activate user account (Use Case)
     */
    public void activateUser(Long userId) {
        User user = userRepository.findById(userId)
            .orElseThrow(() -> new UserNotFoundException("User not found: " + userId));

        user.activate();
        userRepository.update(user);
    }
}
```

```java
package com.bulldog.adapters.out.persistence.mappers;

import com.bulldog.domain.entities.User;
import com.bulldog.domain.value_objects.Email;
import com.bulldog.domain.value_objects.Password;
import com.bulldog.adapters.out.persistence.entities.UserEntity;
import org.springframework.stereotype.Component;

/**
 * Mapper between User Domain Entity and UserEntity (JPA)
 */
@Component
public class UserEntityMapper {

    /**
     * Convert UserEntity (JPA) to User (Domain)
     */
    public User toDomain(UserEntity entity) {
        if (entity == null) {
            return null;
        }

        return User.reconstruct(
            entity.getId(),
            entity.getName(),
            Email.of(entity.getEmail()),
            Password.of(entity.getPassword()),
            entity.getAge(),
            entity.getIsActive(),
            entity.getCreatedAt(),
            entity.getUpdatedAt(),
            null // Bank accounts handled separately
        );
    }

    /**
     * Convert User (Domain) to UserEntity (JPA)
     */
    public UserEntity toEntity(User domainUser) {
        if (domainUser == null) {
            return null;
        }

        return UserEntity.builder()
            .id(domainUser.getId())
            .name(domainUser.getName())
            .email(domainUser.getEmail().value())
            .password(domainUser.getPassword().getHashedValue())
            .age(domainUser.getAge())
            .isActive(domainUser.getIsActive())
            .createdAt(domainUser.getCreatedAt())
            .updatedAt(domainUser.getUpdatedAt())
            .build();
    }
}
```

```java
// filepath: bulldog-api/src/main/java/com/bulldog/adapters/out/persistence/repositories/UserRepositoryImpl.java

package com.bulldog.adapters.out.persistence.repositories;

import com.bulldog.domain.entities.User;
import com.bulldog.domain.repositories.UserRepository;
import com.bulldog.domain.value_objects.Email;
import com.bulldog.adapters.out.persistence.entities.UserEntity;
import com.bulldog.adapters.out.persistence.mappers.UserEntityMapper;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Repository;
import java.util.Optional;
import java.util.List;
import java.util.stream.Collectors;

/**
 * User Repository Implementation (Output Adapter)
 * Implements domain port with JPA
 */
@Repository
@RequiredArgsConstructor
public class UserRepositoryImpl implements UserRepository {

    private final UserJpaRepository jpaRepository;
    private final UserEntityMapper entityMapper;

    @Override
    public User save(User user) {
        UserEntity entity = entityMapper.toEntity(user);
        UserEntity savedEntity = jpaRepository.save(entity);
        return entityMapper.toDomain(savedEntity);
    }

    @Override
    public void update(User user) {
        UserEntity entity = entityMapper.toEntity(user);
        jpaRepository.save(entity);
    }

    @Override
    public Optional<User> findById(Long id) {
        return jpaRepository.findById(id)
            .map(entityMapper::toDomain);
    }

    @Override
    public Optional<User> findByEmail(Email email) {
        return jpaRepository.findByEmail(email.value())
            .map(entityMapper::toDomain);
    }

    @Override
    public List<User> findAll() {
        return jpaRepository.findAll().stream()
            .map(entityMapper::toDomain)
            .collect(Collectors.toList());
    }

    @Override
    public List<User> findAllActive() {
        return jpaRepository.findAllActive().stream()
            .map(entityMapper::toDomain)
            .collect(Collectors.toList());
    }

    @Override
    public boolean existsByEmail(Email email) {
        return jpaRepository.existsByEmail(email.value());
    }

    @Override
    public void deleteById(Long id) {
        jpaRepository.deleteById(id);
    }

    @Override
    public long count() {
        return jpaRepository.count();
    }
}
```

```java
// filepath: bulldog-api/src/main/java/com/bulldog/adapters/out/security/UserPasswordEncoderImpl.java

package com.bulldog.adapters.out.security;

import com.bulldog.application.ports.UserPasswordEncoder;
import com.bulldog.domain.value_objects.Password;
import com.bulldog.domain.exceptions.InvalidPasswordException;
import lombok.RequiredArgsConstructor;
import org.springframework.security.crypto.password.PasswordEncoder;
import org.springframework.stereotype.Component;

/**
 * User Password Encoder Implementation
 * Uses Spring Security BCrypt
 */
@Component
@RequiredArgsConstructor
public class UserPasswordEncoderImpl implements UserPasswordEncoder {

    private final PasswordEncoder passwordEncoder;

    @Override
    public Password encode(String rawPassword) {
        if (rawPassword == null || rawPassword.isBlank()) {
            throw new InvalidPasswordException("Raw password cannot be null or empty");
        }

        String hashedPassword = passwordEncoder.encode(rawPassword);
        return Password.of(hashedPassword);
    }

    @Override
    public boolean matches(String rawPassword, String hashedPassword) {
        if (rawPassword == null || hashedPassword == null) {
            return false;
        }

        return passwordEncoder.matches(rawPassword, hashedPassword);
    }
}
```
